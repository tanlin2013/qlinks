def fidelity_matrix(M, N, normalize=False, tol=1e-15):
    """
    M, N: shape (d, k_M) and (d, k_N) with vectors as columns.
    Returns F of shape (k_M, k_N) with F[i,j] = |<M_i | N_j>|^2.
    """
    M = np.asarray(M, dtype=complex)
    N = np.asarray(N, dtype=complex)

    if normalize:
        # L2-normalize columns; avoid division by ~0
        M_norms = np.linalg.norm(M, axis=0)
        N_norms = np.linalg.norm(N, axis=0)
        M = M / np.maximum(M_norms, tol)
        N = N / np.maximum(N_norms, tol)

    overlaps = M.conj().T @ N          # shape (k_M, k_N)
    F = np.abs(overlaps)**2
    return F

def set_dynamic_cycle(ax, n):
    colors = plt.cm.tab10_r(np.linspace(0, 1, n))
    # Repeating style lists to length n
    markers    = ["*", "x", "v", "^", "o", "d"]
    linestyles = ["-", "--", "-.", ":", "-", "--"]
    # Trim or repeat to match n
    markers    = (markers * ((n // len(markers)) + 1))[:n]
    linestyles = (linestyles * ((n // len(linestyles)) + 1))[:n]
    ax.set_prop_cycle(color=colors, marker=markers, linestyle=linestyles)
